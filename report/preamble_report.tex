%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%% Report-specific preamble
%% Packages and formatting for the technical report build
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%% Page layout
\usepackage[margin=1in]{geometry}
\usepackage{setspace}
\setstretch{1.15}

%% Headers and footers
\usepackage{fancyhdr}
\setlength{\headheight}{14pt}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small Wind Disruption Effects on Overwintering Monarch Clusters}
\fancyhead[R]{\small\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\fancypagestyle{plain}{%
    \fancyhf{}%
    \fancyfoot[C]{\small\thepage}%
    \renewcommand{\headrulewidth}{0pt}%
}

%% Font encoding
\usepackage[T1]{fontenc}

%% Math packages
\usepackage{amsmath}
\usepackage{amssymb}
\usepackage{mathtools}
\usepackage{bm}

%% Figure and table packages
\usepackage{caption}
\usepackage{subcaption}
\captionsetup{
    format=hang,
    font={small, singlespacing},
    labelfont=bf,
    labelsep=colon
}
\captionsetup[table]{position=above}
\captionsetup[subfigure]{labelformat=parens, font={small, singlespacing}}
\captionsetup[subtable]{labelformat=parens, font={small, singlespacing}, position=top}

\usepackage{floatrow}
\floatsetup[table]{capposition=top}

\usepackage{booktabs}
\usepackage{tabularx}
\usepackage{graphicx}
\graphicspath{{figures/}}

%% Long tables (used in appendices)
\usepackage{longtable}

%% TikZ
\usepackage{tikz}
\usetikzlibrary{calc}

%% Bibliography
\usepackage{etoolbox}
\usepackage[
    backend=biber,
    style=apa,
    url=true,
    giveninits=true,
    maxnames=4,
    minnames=3,
]{biblatex}
\patchcmd{\bibsetup}{\interlinepenalty=5000}{\interlinepenalty=10000}{}{}
\AtEveryBibitem{\clearfield{annotation}}
\AtEveryBibitem{\clearlist{language}}
\AtEveryBibitem{\clearfield{note}}
\addbibresource{bibliography/Thesis.bib}

%% URL and hyperlink handling
\usepackage{url}
\usepackage[breaklinks=true,hidelinks,pdfusetitle,hypertexnames=false]{hyperref}
\hypersetup{
    bookmarksnumbered=true,
    bookmarksopen=false,
    colorlinks=false,
    pdfstartview=Fit,
    pdfborder={0 0 0},
}
\urlstyle{same}

%% Allow URLs to break at any character (prevents overflow in bibliography)
\setcounter{biburllcpenalty}{7000}
\setcounter{biburlucpenalty}{8000}
\setcounter{biburlnumpenalty}{7000}

%% Cross-references (must come after hyperref)
\usepackage{cleveref}
\newcommand{\creflastconjunction}{, and~}

%% Figure and table numbering by section
\counterwithin{figure}{section}
\counterwithin{table}{section}

%% Section demotion infrastructure
%% Save original section commands before any \let overrides
\let\realsection\section
\let\realsubsection\subsection
\let\realsubsubsection\subsubsection

%% Provide \chapter command for article class
%% Appendix files from the thesis use \chapter{}
\newcommand{\chapter}[2][]{\clearpage\realsection{#2}}
